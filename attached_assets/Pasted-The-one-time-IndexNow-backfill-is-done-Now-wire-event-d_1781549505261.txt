const LOCALES = ["it", "es", "de", "fr", "pt-br"];

// Canonical public page URL(s) for a review. `locales` = the locales this
// review has translations for (the same set the sitemap emits alternates for).
export function reviewUrls(slug: string, locales: string[] = []): string[] {
  const urls = [`${HOST}/review/${slug}`];
  for (const loc of locales) if (LOCALES.includes(loc)) urls.push(`${HOST}/${loc}/review/${slug}`);
  return urls;
}

Step 3 - Submit module. Create artifacts/api-server/src/indexnow.ts:
tsconst ENDPOINT = "https://api.indexnow.org/indexnow";
const HOST = "cryptokiller.org";

// Returns the endpoint status (200/202 = accepted), or null if skipped/failed.
// Swallows all errors internally - safe to call without try/catch.
export async function submitUrls(urls: string[]): Promise<number | null> {
  const key = process.env.INDEXNOW_KEY;
  if (!key) { console.warn("[indexnow] INDEXNOW_KEY not set; skipping"); return null; }
  const list = Array.from(new Set(urls)).filter((u) => {
    try { const x = new URL(u); return x.protocol === "https:" && x.host === HOST; } catch { return false; }
  });
  if (list.length === 0) return null;
  try {
    const res = await fetch(ENDPOINT, {
      method: "POST",
      headers: { "Content-Type": "application/json; charset=utf-8" },
      body: JSON.stringify({ host: HOST, key, keyLocation: `https://${HOST}/${key}.txt`, urlList: list }),
    });
    console.log(`[indexnow] ${list.length} url(s) -> ${res.status} ${res.statusText}: ${list.join(", ")}`);
    return res.status;
  } catch (err) {
    console.error("[indexnow] submit failed (non-fatal):", err);
    return null;
  }
}

// Fire-and-forget. Never awaited, never blocks or throws into the request lifecycle.
export function pingIndexNow(urls: string[]): void {
  if (urls && urls.length) void submitUrls(urls);
}
